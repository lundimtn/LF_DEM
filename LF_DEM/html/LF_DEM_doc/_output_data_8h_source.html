40  int getLineNumber()
41  {
42  unsigned int line_nb = 0;
43  for (const auto& data : output_data) {
44  const auto &col = data.second;
45  if (line_nb == 0 && col.size() > 0) {
46  line_nb = col.size();
47  }
48  if (col.size() > 0 && col.size() != line_nb) {
49  std::cerr << " Error: inconsistent output. Number of lines to output is heterogeneous." << std::endl;
50  exit(1);
51  }
52  }
53  return line_nb;
54  }

56  void initCol(std::string name,
57  std::string physical_dimension,
58  int width)
59  {
60  if (output_data.find(name) != output_data.end()) {
61  return;
62  }
63  std::vector <std::string> col;
64  col.clear();
65  output_data[name] = col;
66  output_data_name.push_back(name);
67  output_data_type[name] = physical_dimension;
68  output_data_width[name] = width;
69  insert_order.push_back(name);
70  }

107  void setDimensionlessNumber(double dimensionless_number)
108  // dimensionless_number = internal_force_unit/output_force_unit
109  {
110  if (dimensionless_number == 0) {
111  std::cerr << "dimensionless_number (internal_force_unit/output_force_unit) = " << dimensionless_number << std::endl;
112  dimensionless_number = 1; // @@@@ To be checked.
113  }
114  converter["none"] = 1;
115  converter["viscosity"] = 6*M_PI;
116  converter["stress"] = 6*M_PI*dimensionless_number;
117  converter["force"] = dimensionless_number;
118  converter["time"] = 1/dimensionless_number;
119  converter["rate"] = dimensionless_number;
120  converter["velocity"] = dimensionless_number;
121  }

123  template<typename T>
124  void entryData(std::string name,
125  std::string physical_dimension,
126  int width,
127  T value,
128  int precision=-1)
129  {
130  if (first_time) {
131  initCol(name, physical_dimension, width);
132  }
133  int output_precision;
134  if (precision > 0) {
135  output_precision = precision;
136  } else {
137  output_precision = default_precision;
138  }
139  std::ostringstream str_value;
140  str_value << std::setprecision(output_precision) << converter[output_data_type[name]]*value;
141  output_data[name].push_back(str_value.str());
142  }

144  void writeFileHeader() {
145  fout << "# data in " << out_unit << " units." << std::endl;
146  int i = 1;
147  for (const auto &name : insert_order) {
148  int width = output_data_width[name];
149  if (width == 1) {
150  fout << "#" << i << ": ";
151  } else {
152  fout << "#" << i << "-" << i+width-1 << ": ";
153  }
154  i += width;
155  fout << name;
156  fout << std::endl;
157  }
158  fout << std::endl;
159  }
160 
161  void writeColsToFile() {
162  int line_nb = getLineNumber();
163  for (int i=0; i<line_nb; i++) {
164  for (const auto& name : insert_order) {
165  const auto &col = output_data[name];
166  if (!col.empty()) {
167  fout << col[i] << " ";
168  } else {
169  fout << "n ";
170  }
171  }
172  fout << std::endl;
173  }
174  for (auto& od : output_data) {
175  od.second.clear();
176  }
177  }
