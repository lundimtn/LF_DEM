28  /*******************************************************
29  HYDRODYNAMICS
30  ********************************************************/
31  /*
32  * Stokes drag coeffient
33  */
34  /* sd_coeff:
35  * Full Stokes drag is given by sd_coeff = 1.
36  * sd_coeff = 0 makes the resistance matrix singular.
37  * sd_coeff = 1e-3 may be reasonable to remove the effect of the drag
38  */
39  double sd_coeff;
40  /*
41  * Lubrication model
42  * 0 no lubrication (no dashpot as well)
43  * 1 1/xi lubrication (only squeeze mode)
44  * 2 log(1/xi) lubrication
45  * 3 dashpot in the contact model (not implemented yet)
46  * In this case lub_max_gap should be automatically zero. (not yet implemented)
47  */
48  std::string lubrication_model;
49  /*
50  * Leading term of lubrication force is 1/reduced_gap,
51  * with reduced_gap the gap
52  * reduced_gap = 2r/(a0+a1) - 2.
53  * we set a cutoff for the lubrication interaction,
54  * such that the lub term is proportional to:
55  *
56  * 1/(reduced_gap+lub_reduce_parameter)
57  * when reduced_gap > 0.
58  */
59  double lub_reduce_parameter;
60  double lub_max_gap;
61  /*******************************************************
62  CONTACTS
63  ********************************************************/
64  /*
65  * 0 No friction
66  * 1 Linear friction law Ft < mu Fn
67  * 2 Threshold friction without repulsive force
68  */
69  double friction_model;
70  double mu_static;
71  double mu_dynamic;
72  double mu_rolling;
73  double ft_max;
74  /*
75  * Contact force parameters
76  * kn: normal spring constant
77  * kt: tangential spring constant
78  */
79  double kn;
80  double kt;
81  double kr;
82  /*
83  * contact_relaxation_factor:
84  *
85  * This gives the coeffient of the resistance term for h < 0.
86  * - If the value is negative, the value of 1/lub_reduce_parameter is used.
87  *
88  */
89  double contact_relaxation_time;
90  double contact_relaxation_time_tan;
