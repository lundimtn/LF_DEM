29  /*******************************************************
30  HYDRODYNAMICS
31  ********************************************************/
32  /*
33  * Stokes drag coeffient
34  */
35  /* sd_coeff:
36  * Full Stokes drag is given by sd_coeff = 1.
37  * sd_coeff = 0 makes the resistance matrix singular.
38  * sd_coeff = 1e-3 may be reasonable to remove the effect of the drag
39  */
40  double sd_coeff;
41  /*
42  * Lubrication model
43  * 0 no lubrication
44  * 1 1/xi lubrication (only squeeze mode)
45  * 2 log(1/xi) lubrication
46  * 3 ???
47  */
48  int lubrication_model;
49  /*
50  * Leading term of lubrication force is 1/reduced_gap,
51  * with reduced_gap the gap
52  * reduced_gap = 2r/(a0+a1) - 2.
53  * we set a cutoff for the lubrication interaction,
54  * such that the lub term is proportional to:
55  *
56  * 1/(reduced_gap+lub_reduce_parameter)
57  * when reduced_gap > 0.
58  */
59  double lub_reduce_parameter;
60  double lub_max_gap;
61 
62  /*******************************************************
63  CONTACTS
64  ********************************************************/
65  /*
66  * 0 No friction
67  * 1 Linear friction law Ft < mu Fn
68  * 2 Threshold friction without repulsive force
69  * 3 Threshold friction without repulsion + mu inf
70  */
71  double friction_model;
72  double mu_static;
73  double mu_dynamic;
74  double mu_rolling;
75  double ft_max;
76  /*
77  * Contact force parameters
78  * kn: normal spring constant
79  * kt: tangential spring constant
80  */
81  bool unscaled_contactmodel;
82  double kn;
83  double kt;
84  double kr;
85  /*
86  * contact_relaxation_factor:
87  *
88  * This gives the coeffient of the resistance term for h < 0.
89  * - If the value is negative, the value of 1/lub_reduce_parameter is used.
90  *
91  */
92  double contact_relaxation_time;
93  double contact_relaxation_time_tan;
94 
95  /*******************************************************
