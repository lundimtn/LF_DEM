38  /*
39  * Stokes drag coeffient
40  */
41  /* sd_coeff:
42  * Full Stokes drag is given by sd_coeff = 1.
43  * sd_coeff = 0 makes the resistance matrix singular.
44  * sd_coeff = 1e-3 may be reasonable to remove the effect of the drag
45  */
46  double sd_coeff;
47  /*
48  * Lubrication model
49  * 0 no lubrication
50  * 1 1/xi lubrication (only squeeze mode)
51  * 2 log(1/xi) lubrication
52  * 3 ???
53  */
54  int lubrication_model;
55  /*
56  * Leading term of lubrication force is 1/reduced_gap,
57  * with reduced_gap the gap
58  * reduced_gap = 2r/(a0+a1) - 2.
59  * we set a cutoff for the lubrication interaction,
60  * such that the lub term is proportional to:
61  *
62  * 1/(reduced_gap+lub_reduce_parameter)
63  * when reduced_gap > 0.
64  */
65  double lub_reduce_parameter;
66  double lub_max_gap;
